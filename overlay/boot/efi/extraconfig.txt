# uConsole CM5 Stabilization - Master Config (March 2026)

# 1. Disable conflicting standard controllers early
# Disable SPI to prevent GPIO 1 collision with PMIC I2C
dtparam=spi=off

# 2. Base Overlays (Required for CM5 hardware support)
dtoverlay=clockworkpi-uconsole-cm5-stable
dtoverlay=uconsole-audio
dtoverlay=uconsole-sd-alt

# 3. USB Configuration
dtoverlay=dwc2,dr_mode=host

# 4. Display Timings & TTY Fix
# Removal of video= override from GRUB allows panel driver safe 40MHz clock.
# usbcore.autosuspend=-1 prevents keyboard sleep issues.
cmdline=usbcore.autosuspend=-1 fbcon=rotate:1

# 5. Boot Pins (Force backlight high at boot)
gpio=9=op,dh
gpio=10=ip,np
